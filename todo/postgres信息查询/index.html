Ansible部署系统监控功能

Ansible postgresql管理模块

postgresql_copy – Copy data between a file/program and a PostgreSQL table postgresql_db – Add or remove PostgreSQL databases from a remote host postgresql_ext – Add or remove PostgreSQL extensions from a database postgresql_idx – Create or drop indexes from a PostgreSQL database postgresql_info – Gather information about PostgreSQL servers postgresql_lang – Adds, removes or changes procedural languages with a PostgreSQL database postgresql_membership – Add or remove PostgreSQL roles from groups postgresql_owner – Change an owner of PostgreSQL database object postgresql_ping – Check remote PostgreSQL server availability postgresql_privs – Grant or revoke privileges on PostgreSQL database objects postgresql_publication – Add, update, or remove PostgreSQL publication postgresql_query – Run PostgreSQL queries postgresql_schema – Add or remove PostgreSQL schema postgresql_sequence – Create, drop, or alter a PostgreSQL sequence postgresql_set – Change a PostgreSQL server configuration parameter postgresql_slot – Add or remove replication slots from a PostgreSQL database postgresql_table – Create, drop, or modify a PostgreSQL table postgresql_tablespace – Add or remove PostgreSQL tablespaces from remote hosts postgresql_user – Add or remove a user (role) from a PostgreSQL server instance

Ansible Tower Postgres信息查询语句

获取客户机IP地址列表

select distinct
  ansible_facts#>>'{ansible_default_ipv4, address}' as ipv4
from (
    select distinct
      ansible_facts
    from main_host
    where not ansible_facts = '{}'
  ) ANSIBLE_FACTS;

获取部署仓库代码列表

select
  scm_type as 类型,
  scm_url as 仓库地址,
  scm_branch as 分支,
  substring(scm_revision, 0, 11) as 版本,
  playbook_files as 部署脚本列表
from
  main_project

获取仓库版本列表

select
    substring(scm_revision, 0, 11) as revision 
from  
    main_project
where
    not scm_revision = '' and scm_url like '%$project%'

获取仓库地址

select
  scm_url as url
from
  main_project
where
  scm_revision like '%$revision%'
and
  scm_url like '%$project%'

获取仓库分支

select
  scm_url as url
from
  main_project
where
  scm_revision like '%$revision%'
and
  scm_url like '%$project%'

获取客户机CPU信息

select
  regexp_split_to_table(
    regexp_replace(
          regexp_replace(trim(both E'[\]\]",]' from ansible_facts->>'ansible_processor'), E'("\\d+)",\\s+"', '~@~\1 ', 'g'),
    E'[",\\s+",]', ' ', 'g'),
  E'~@~')  
  as 处理器
from
  (
    select distinct
      ansible_facts,
      modified
    from
      main_host
    where
      ansible_facts#>>'{ansible_default_ipv4, address}'  = '$ip' and
      not ansible_facts = '{}'
    order by modified desc
    limit 1
  ) ANSIBLE_FACTS

获取内存地址

select
  mem_info->>'key' as 类型,
  mem_info#>'{value, free}' as 剩余,
  mem_info#>'{value, used}' as 已使用,
  mem_info#>'{value, total}' as 总量
from
(select
  row_to_json(jsonb_each(ansible_facts->'ansible_memory_mb')) as mem_info
from (
    select distinct
      ansible_facts,
      modified
    from main_host
    where 
      ansible_facts#>>'{ansible_default_ipv4, address}' = '$ip' and
      not ansible_facts = '{}'
    order by modified desc
    limit 1
  ) ANSIBLE_FACTS)ANSIBLE_DEVICES

获取客户机IP地址列表

select distinct
  regexp_split_to_table(regexp_replace(trim(both E'[]' from ansible_facts->>'ansible_all_ipv4_addresses'), E'"','','g'), E',') as ip地址列表
from (
    select distinct
      ansible_facts,
      modified
    from main_host
    where
      ansible_facts#>>'{ansible_default_ipv4, address}'  = '$ip' and
      not ansible_facts = '{}'
    order by modified desc
    limit 1
  ) ANSIBLE_FACTS;

获取客户机磁盘信息列表

select
  disk_info::jsonb->>'device' as device,
  disk_info::jsonb->>'fstype' as fstype,
  disk_info::jsonb->>'size_total' as total_size,
  disk_info::jsonb->>'size_available' as available_size
from
(select
  jsonb_array_elements(ansible_facts->'ansible_mounts') as disk_info
from (
    select distinct
      ansible_facts,
      modified
    from main_host
    where
        ansible_facts#>>'{ansible_default_ipv4, address}'  = '$ip' and
        not ansible_facts = '{}'
    order by modified desc
    limit 1
  ) ANSIBLE_FACTS)ANSIBLE_DISKS

获取客户机设备信息

select
  disk_info->>'key' as 设备名,
  disk_info#>>'{value, size}' as 大小,
  disk_info#>>'{value, model}' as 类型,
  disk_info#>>'{value, host}' as 信息
from
(select
  row_to_json(jsonb_each(ansible_facts->'ansible_devices')) as disk_info
from (
    select distinct
      ansible_facts,
      modified
    from main_host
    where
        ansible_facts#>>'{ansible_default_ipv4, address}'  = '$ip'  and
        not ansible_facts = '{}'
    order by modified desc
    limit 1
  ) ANSIBLE_FACTS)ANSIBLE_DEVICES

获取硬盘大小

select
  disk_info->>'key' as 设备名,
  disk_info#>>'{value, size}' as 大小
from
(select
  row_to_json(jsonb_each(ansible_facts->'ansible_devices')) as disk_info
from (
    select distinct
      ansible_facts,
      modified
    from main_host
    where
        ansible_facts#>>'{ansible_default_ipv4, address}' = '$ip' and 
        not ansible_facts = '{}'
    order by modified desc
    limit 1
  ) ANSIBLE_FACTS)ANSIBLE_DEVICES
where
  (disk_info->>'key') ~ 'sd'
and
  disk_info#>>'{value, model}' <> ''
